"""
Comprehensive remote-access hardening for headless mini PC:
1. Disable UAC consent prompt (no more screen dimming)
2. Set Windows auto-login with password
3. Ensure RustDesk service config has password-only auth
4. Disable lock screen timeout / screen saver lock
5. Power settings: never sleep
6. Headless display: keep virtual display alive when no monitor
7. Prevent display driver from disabling on monitor disconnect
"""
import subprocess, os, sys

def run(cmd, desc, shell=True):
    try:
        r = subprocess.run(cmd, shell=shell, capture_output=True, text=True, timeout=30)
        status = "OK" if r.returncode == 0 else "WARN"
        print(f"  [{status}] {desc}")
        if r.stdout.strip():
            print(f"       {r.stdout.strip()[:200]}")
        if r.returncode != 0 and r.stderr.strip():
            print(f"       ERR: {r.stderr.strip()[:200]}")
        return r.returncode == 0
    except Exception as e:
        print(f"  [FAIL] {desc}: {e}")
        return False

print("=" * 60)
print("  HARDEN REMOTE ACCESS — ANTI-LOCKOUT")
print("=" * 60)

ok = 0
fail = 0

# === 1. DISABLE UAC CONSENT PROMPT ===
print("\n[1] Disabling UAC consent prompt (no screen dimming)...")

# ConsentPromptBehaviorAdmin=0 means "Elevate without prompting"
# This prevents the secure desktop dimming that blocks RustDesk
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System" /v ConsentPromptBehaviorAdmin /t REG_DWORD /d 0 /f',
       "Set UAC admin: elevate without prompting"):
    ok += 1
else:
    fail += 1

# ConsentPromptBehaviorUser=0 means auto-deny for standard users (safe)
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System" /v ConsentPromptBehaviorUser /t REG_DWORD /d 0 /f',
       "Set UAC user: auto-deny"):
    ok += 1
else:
    fail += 1

# PromptOnSecureDesktop=0 means don't switch to secure desktop
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System" /v PromptOnSecureDesktop /t REG_DWORD /d 0 /f',
       "Disable secure desktop (screen dimming)"):
    ok += 1
else:
    fail += 1

# EnableLUA=1 — keep UAC technically "on" for compatibility, just no prompts
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System" /v EnableLUA /t REG_DWORD /d 1 /f',
       "Keep UAC enabled (for app compatibility)"):
    ok += 1
else:
    fail += 1

# === 2. AUTO-LOGIN ===
print("\n[2] Setting Windows auto-login...")
winlogon = "HKLM\\SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Winlogon"

if run(f'reg add "{winlogon}" /v AutoAdminLogon /t REG_SZ /d 1 /f',
       "AutoAdminLogon = 1"):
    ok += 1
else:
    fail += 1

if run(f'reg add "{winlogon}" /v DefaultUserName /t REG_SZ /d C /f',
       "DefaultUserName = C"):
    ok += 1
else:
    fail += 1

if run(f'reg add "{winlogon}" /v DefaultPassword /t REG_SZ /d CMCPassTemp7508! /f',
       "DefaultPassword = (set)"):
    ok += 1
else:
    fail += 1

# Remove any domain name requirement
run(f'reg add "{winlogon}" /v DefaultDomainName /t REG_SZ /d "" /f',
    "Clear DefaultDomainName")

# === 3. DISABLE LOCK SCREEN / SCREEN SAVER LOCK ===
print("\n[3] Preventing screen lock on idle...")

# Disable screen saver
if run('reg add "HKCU\\Control Panel\\Desktop" /v ScreenSaveActive /t REG_SZ /d 0 /f',
       "Disable screen saver"):
    ok += 1
else:
    fail += 1

# Don't require password on wake
if run('reg add "HKCU\\Control Panel\\Desktop" /v ScreenSaverIsSecure /t REG_SZ /d 0 /f',
       "Disable password on wake"):
    ok += 1
else:
    fail += 1

# Disable lock screen
if run('reg add "HKLM\\SOFTWARE\\Policies\\Microsoft\\Windows\\Personalization" /v NoLockScreen /t REG_DWORD /d 1 /f',
       "Disable lock screen"):
    ok += 1
else:
    fail += 1

# Disable Ctrl+Alt+Del requirement
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System" /v DisableCAD /t REG_DWORD /d 1 /f',
       "Disable Ctrl+Alt+Del requirement"):
    ok += 1
else:
    fail += 1

# Disable automatic lock via group policy
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System" /v InactivityTimeoutSecs /t REG_DWORD /d 0 /f',
       "Disable inactivity lock timeout"):
    ok += 1
else:
    fail += 1

# Prevent Windows from requiring sign-in after sleep
if run('reg add "HKLM\\SOFTWARE\\Policies\\Microsoft\\Power\\PowerSettings\\0e796bdb-100d-47d6-a2d5-f7d2daa51f51" /v ACSettingIndex /t REG_DWORD /d 0 /f',
       "No sign-in after sleep (AC)"):
    ok += 1
else:
    fail += 1

# === 4. POWER SETTINGS — NEVER SLEEP ===
print("\n[4] Preventing sleep/hibernate...")

# Set power plan to never sleep
run('powercfg /change standby-timeout-ac 0', "Never sleep on AC")
run('powercfg /change monitor-timeout-ac 0', "Never turn off display on AC")
run('powercfg /change hibernate-timeout-ac 0', "Never hibernate on AC")
run('powercfg /h off', "Disable hibernation entirely")
ok += 4  # These usually succeed

# === 5. VERIFY RUSTDESK CONFIG ===
print("\n[5] Verifying RustDesk service config...")

svc_config = r"C:\Windows\ServiceProfiles\LocalService\AppData\Roaming\RustDesk\config\RustDesk2.toml"
usr_config = os.path.expandvars(r"%APPDATA%\RustDesk\config\RustDesk2.toml")

for label, path in [("service", svc_config), ("user", usr_config)]:
    if os.path.exists(path):
        with open(path, 'r') as f:
            content = f.read()
        has_pw = 'use-permanent-password' in content
        has_approval = 'approval-mode' in content
        print(f"  [{label}] verification-method: {'OK' if has_pw else 'MISSING'}")
        print(f"  [{label}] approval-mode: {'present' if has_approval else 'not set'}")

        # Fix if missing
        needs_write = False
        if 'use-permanent-password' not in content:
            if '[options]' in content:
                content = content.replace('[options]', "[options]\nverification-method = 'use-permanent-password'")
                needs_write = True
            else:
                content += "\n[options]\nverification-method = 'use-permanent-password'\n"
                needs_write = True

        if "approval-mode = 'password'" not in content:
            if '[options]' in content:
                content = content.replace('[options]', "[options]\napproval-mode = 'password'")
                needs_write = True

        if needs_write:
            try:
                with open(path, 'w') as f:
                    f.write(content)
                print(f"  [{label}] FIXED config")
                ok += 1
            except PermissionError:
                print(f"  [{label}] Cannot write (permission denied)")
                fail += 1
        else:
            print(f"  [{label}] Config OK")
            ok += 1
    else:
        print(f"  [{label}] NOT FOUND at {path}")
        fail += 1

import time

# === 6. VERIFY CURRENT STATE ===
print("\n[6] Pre-restart verification...")

# Check UAC setting
r = subprocess.run('reg query "HKLM\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Policies\\System" /v ConsentPromptBehaviorAdmin',
                    shell=True, capture_output=True, text=True)
if "0x0" in r.stdout:
    print("  [OK] UAC consent prompt disabled (no screen dimming)")
else:
    print("  [WARN] UAC setting may not have applied")

# Check auto-login
r = subprocess.run(f'reg query "{winlogon}" /v DefaultPassword', shell=True, capture_output=True, text=True)
if r.returncode == 0:
    print("  [OK] Auto-login password is set")
else:
    print("  [FAIL] Auto-login password not set")

# Check RustDesk service
r = subprocess.run('sc query RustDesk', shell=True, capture_output=True, text=True)
if "RUNNING" in r.stdout:
    print("  [OK] RustDesk service running")
else:
    print("  [WARN] RustDesk service not running")

# === 8. HEADLESS DISPLAY — KEEP VIRTUAL DISPLAY WHEN NO MONITOR ===
print("\n[8] Configuring headless display (no-monitor operation)...")

# Prevent Windows from disabling display adapter when no monitor detected
# This is critical for RustDesk — without a display, it renders blank/frozen

# Tell GPU driver to always assume a display is connected
# Works for most Intel/AMD integrated GPUs used in mini PCs
if run('reg add "HKLM\\SYSTEM\\CurrentControlSet\\Control\\GraphicsDrivers\\Configuration" /v SimulateMonitor /t REG_DWORD /d 1 /f',
       "Simulate monitor presence in GPU driver"):
    ok += 1
else:
    fail += 1

# Disable display turn-off (redundant with powercfg but belt-and-suspenders)
if run('reg add "HKCU\\Control Panel\\PowerCfg\\GlobalPowerPolicy" /v Policies /t REG_BINARY /d 01000000 /f',
       "Disable display power-off policy"):
    ok += 1
    # This one often fails — not critical
else:
    pass

# Force Windows to keep the desktop compositor (DWM) running even headless
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\Dwm" /v ForceEffectMode /t REG_DWORD /d 2 /f',
       "Force DWM compositor always on"):
    ok += 1
else:
    fail += 1

# Prevent Windows from switching to "basic" display when monitor disconnects
if run('reg add "HKLM\\SOFTWARE\\Microsoft\\Windows\\Dwm" /v EnableMachineCheck /t REG_DWORD /d 0 /f',
       "Prevent DWM from degrading on disconnect"):
    ok += 1
else:
    fail += 1

# Set minimum resolution fallback so RustDesk always has something to capture
if run('reg add "HKLM\\SYSTEM\\CurrentControlSet\\Control\\GraphicsDrivers" /v TdrLevel /t REG_DWORD /d 0 /f',
       "Disable Timeout Detection & Recovery (TDR) for GPU"):
    ok += 1
else:
    fail += 1

# Disable Connected Standby / Modern Standby (which can put headless PCs in weird power states)
if run('reg add "HKLM\\SYSTEM\\CurrentControlSet\\Control\\Power" /v CsEnabled /t REG_DWORD /d 0 /f',
       "Disable Connected Standby"):
    ok += 1
else:
    fail += 1

# Disable fast startup (can cause display issues on headless boot)
if run('reg add "HKLM\\SYSTEM\\CurrentControlSet\\Control\\Session Manager\\Power" /v HiberbootEnabled /t REG_DWORD /d 0 /f',
       "Disable Fast Startup"):
    ok += 1
else:
    fail += 1

# === 9. RUSTDESK HEADLESS OPTIMIZATION ===
print("\n[9] Optimizing RustDesk for headless operation...")

# Check if RustDesk has --headless or display settings in its config
for label, path in [("service", svc_config), ("user", usr_config)]:
    if os.path.exists(path):
        with open(path, 'r') as f:
            content = f.read()
        needs_write = False

        # Ensure allow-remote-config-modification is set (so we can manage remotely)
        if "allow-remote-config-modification" not in content:
            if '[options]' in content:
                content = content.replace('[options]', "[options]\nallow-remote-config-modification = 'Y'")
                needs_write = True

        # Enable direct-server mode for better performance (fewer relays)
        if "direct-server" not in content:
            if '[options]' in content:
                content = content.replace('[options]', "[options]\ndirect-server = 'Y'")
                needs_write = True

        # Enable auto-accept file transfer (useful for command runner)
        if "enable-file-transfer" not in content:
            if '[options]' in content:
                content = content.replace('[options]', "[options]\nenable-file-transfer = 'Y'")
                needs_write = True

        if needs_write:
            try:
                with open(path, 'w') as f:
                    f.write(content)
                print(f"  [{label}] Added headless optimizations")
                ok += 1
            except PermissionError:
                print(f"  [{label}] Permission denied writing config")
                fail += 1
        else:
            print(f"  [{label}] Already configured")
            ok += 1

print("\n[11] Final RustDesk restart...")
run('net stop RustDesk', "Stop RustDesk service")
time.sleep(2)
run('net start RustDesk', "Start RustDesk service")

print(f"\n{'=' * 60}")
print(f"  RESULTS: {ok} succeeded, {fail} failed")
print(f"  UAC: Consent prompts disabled (no screen dimming)")
print(f"  Auto-login: Configured for user C")
print(f"  RustDesk: Password-only (no manual approve needed)")
print(f"  Display: Never locks, never sleeps, never turns off")
print(f"  Headless: Virtual display kept alive, DWM forced on")
print(f"  Power: No sleep, no hibernate, no Connected Standby")
