Moves everything else into categorized subdirectories.
Updates all path references in scripts and scheduled tasks.

import os, shutil, subprocess, glob

DESKTOP = r"C:\Users\C\Desktop"
SCRIPTS_RUDY = os.path.join(DESKTOP, "scripts", "rudy")
SCRIPTS_WH = os.path.join(DESKTOP, "scripts", "workhorse")

# Ensure all directories exist
for d in [SCRIPTS_RUDY, SCRIPTS_WH, SCRIPTS_ARCHIVE, DOCS, DATA_SESSIONS, DATA_BROWSER, CMD_ARCHIVE]:
    os.makedirs(d, exist_ok=True)

print("=" * 60)

moved = 0
errors = 0

def safe_move(src, dst_dir, label=""):
    global moved, errors
    fname = os.path.basename(src)
    dst = os.path.join(dst_dir, fname)
    if not os.path.exists(src):
        return
    if os.path.exists(dst):
        # Don't overwrite — skip
        print(f"  SKIP (exists): {fname}")
        return
    try:
        shutil.move(src, dst)
        print(f"  MOVED: {fname} → {label or os.path.basename(dst_dir)}/")
        moved += 1
    except Exception as e:
        print(f"  ERROR: {fname}: {e}")
        errors += 1
